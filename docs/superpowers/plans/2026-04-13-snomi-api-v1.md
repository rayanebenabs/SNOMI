# SNOMI API v1 — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a TypeScript + Hono REST API exposing 4 SNOMI endpoints (decode, benchmark, compare, audit) backed by static JSON data compiled from the existing spec Markdown files.

**Architecture:** `scripts/compile-specs.ts` parses `references/*.md` → `src/data/*.json` once. At startup, `src/lib/snomi-index.ts` loads these JSON files and builds in-memory Maps for O(1) lookups. Four Hono route files handle the endpoints. No database, no auth.

**Tech Stack:** Node.js 20+, TypeScript 5, Hono 4, @hono/node-server, Vitest 2, tsx

```typescript
import { readFileSync, writeFileSync, mkdirSync } from 'node:fs'
import { join } from 'node:path'

const REFS = join(process.cwd(), '..', 'references')
const OUT = join(process.cwd(), 'src', 'data')

mkdirSync(OUT, { recursive: true })

// ── Helpers ──────────────────────────────────────────────────────────────────

function parseTableRows(lines: string[]): Array<Record<string, string>> {
  const tableLines = lines.filter(l => l.startsWith('|') && !l.match(/^\|[\s\-|]+\|/))
  if (tableLines.length < 2) return []
  const headers = tableLines[0]
    .split('|').slice(1, -1)
    .map(h => h.trim().toLowerCase()
      .replace(/\s+\(codified\)/, '')
      .replace(/\s+/g, '_')
      .replace(/[^a-z0-9_]/g, ''))
  return tableLines.slice(1).map(row => {
    const cells = row.split('|').slice(1, -1).map(c => c.trim())
    return Object.fromEntries(headers.map((h, i) => [h, cells[i] ?? '']))
  })
}

// ── metrics.json ──────────────────────────────────────────────────────────────

function compileMetrics() {
  const content = readFileSync(join(REFS, 'metrics.md'), 'utf-8')
  const lines = content.split('\n')

  const results: object[] = []
  let domain = ''
  let tableBuffer: string[] = []

  function flushTable() {
    if (!tableBuffer.length) return
    const rows = parseTableRows(tableBuffer)
    for (const r of rows) {
      const code = (r.code ?? '').trim()
      if (!code.match(/^[A-Z]{2,3}\.\d{3}/)) continue
      results.push({
        code,
        domain,
        name: r.name ?? '',
        definition: r.definition ?? '',
        formula: r.formula_codified ?? r.formula ?? '',
        unit: r.format ?? '',
        trust_level: (r.trust ?? 'T3').split('/')[0].trim(),
        notes: null,
      })
    }
    tableBuffer = []
  }

  for (const line of lines) {
    const domainMatch = line.match(/^## (AUD|ENG|REA|CON|VAL|QUA|CRE|REG)\b/)
    if (domainMatch) { flushTable(); domain = domainMatch[1] }
    if (line.startsWith('|')) {
      tableBuffer.push(line)
    } else if (tableBuffer.length && !line.startsWith('|') && line.trim()) {
      flushTable()
    }
  }
  flushTable()

  // Deduplicate by code
  const seen = new Set<string>()
  const unique = results.filter((m: any) => !seen.has(m.code) && seen.add(m.code))

  writeFileSync(join(OUT, 'metrics.json'), JSON.stringify(unique, null, 2))
  console.log(`✓ metrics.json — ${unique.length} entries`)
}

// ── benchmarks.json ───────────────────────────────────────────────────────────

const NICHE_LABELS: Record<string, string> = {
  'Fashion & Beauty': 'fashion_beauty',
  'Food & Lifestyle': 'food_lifestyle',
  'Sports & Fitness': 'sports_fitness',
  'Tech & Gaming': 'tech_gaming',
  'Travel': 'travel',
  'Educational content': 'education',
  'Parenting': 'parenting',
  'Sustainability': 'sustainability',
  'Comedy & Entertainment': 'comedy_entertainment',
  'Beauty & Skincare': 'beauty_skincare',
}

function parsePct(v: string): number {
  return parseFloat(v.replace('%', '').replace(/[^\d.]/g, '')) || 0
}

function compileBenchmarks() {
  const content = readFileSync(join(REFS, 'benchmarks.md'), 'utf-8')
  const sections = content.split(/\n## /).filter(Boolean)
  const entries: object[] = []

  for (const section of sections) {
    const firstLine = section.split('\n')[0].trim()
    const codeMatch = firstLine.match(/^([A-Z]+\.\d{3}\.[A-Z]+)/)
    if (!codeMatch) continue

    const metricCode = codeMatch[1] // e.g. "ENG.002.IG"
    const lines = section.split('\n')
    const tableLines = lines.filter(l => l.startsWith('|') && !l.match(/^\|[\s\-|]+\|/))
    if (tableLines.length < 2) continue

    const headers = tableLines[0].split('|').slice(1, -1).map(h => h.trim())

    for (const row of tableLines.slice(1)) {
      const cells = row.split('|').slice(1, -1).map(c => c.trim())
      const nicheLabel = cells[0] ?? ''
      const niche = NICHE_LABELS[nicheLabel]
      if (!niche) continue

      const regions: Record<string, number> = {}
      const tiers: Record<string, number> = {}

      headers.slice(1).forEach((h, i) => {
        const val = parsePct(cells[i + 1] ?? '0')
        if (h === 'Global') regions.global = val
        else if (h === 'NA') regions.na = val
        else if (h === 'EMEA') regions.emea = val
        else if (h === 'APAC') regions.apac = val
        else if (h === 'LATAM') regions.latam = val
        else if (/Nano/i.test(h)) tiers.nano = val
        else if (/Macro/i.test(h)) tiers.macro = val
      })

      entries.push({ metric: metricCode, niche, regions, tiers })
    }
  }

  writeFileSync(join(OUT, 'benchmarks.json'), JSON.stringify(entries, null, 2))
  console.log(`✓ benchmarks.json — ${entries.length} entries`)
}

// ── valuesets.json ────────────────────────────────────────────────────────────

function compileValueSets() {
  const content = readFileSync(join(REFS, 'foundation/valuesets.md'), 'utf-8')
  const lines = content.split('\n')

  const platforms: string[] = []
  const formats: string[] = []
  let section = ''

  for (const line of lines) {
    if (line.includes('ValueSet/platforms')) section = 'platforms'
    else if (line.includes('ValueSet/content-formats')) section = 'formats'
    else if (line.includes('ValueSet/content-types')) section = ''

    const codeMatch = line.match(/^\| `([A-Z0-9]+)` \|/)
    if (!codeMatch) continue
    if (section === 'platforms') platforms.push(codeMatch[1])
    if (section === 'formats') formats.push(codeMatch[1])
  }

  const valuesets = {
    platforms,
    formats,
    tiers: ['nano', 'micro', 'mid', 'macro', 'mega'],
    country_to_region: {
      US: 'na', CA: 'na',
      GB: 'emea', UK: 'emea', FR: 'emea', DE: 'emea', IT: 'emea',
      ES: 'emea', NL: 'emea', BE: 'emea', SE: 'emea', CH: 'emea',
      JP: 'apac', AU: 'apac', KR: 'apac', IN: 'apac', SG: 'apac',
      BR: 'latam', MX: 'latam', AR: 'latam', CO: 'latam',
    },
  }

  writeFileSync(join(OUT, 'valuesets.json'), JSON.stringify(valuesets, null, 2))
  console.log(`✓ valuesets.json — ${platforms.length} platforms, ${formats.length} formats`)
}

compileMetrics()
compileBenchmarks()
compileValueSets()
```

```typescript
import { readFileSync } from 'node:fs'
import { join, dirname } from 'node:path'
import { fileURLToPath } from 'node:url'
import type { MetricDefinition, BenchmarkEntry, ValueSets, Distribution } from './types.js'

const __dirname = dirname(fileURLToPath(import.meta.url))
const load = <T>(file: string): T =>
  JSON.parse(readFileSync(join(__dirname, '../../data', file), 'utf-8')) as T

const metrics = load<MetricDefinition[]>('metrics.json')
const benchmarks = load<BenchmarkEntry[]>('benchmarks.json')
const valuesets = load<ValueSets>('valuesets.json')

// ── Metrics index ─────────────────────────────────────────────────────────────

const metricIndex = new Map<string, MetricDefinition>()
for (const m of metrics) metricIndex.set(m.code, m)

export function parseCode(code: string) {
  const parts = code.replace(/\[.*?\]/g, '').split('.')
  return {
    base: parts.slice(0, 2).join('.'),
    domain: parts[0] ?? '',
    id: parts[1] ?? '',
    platform: parts[2] ?? null,
    market: parts[3] ?? null,
  }
}

export function lookupMetric(code: string): MetricDefinition | undefined {
  const { base } = parseCode(code)
  return metricIndex.get(base)
}

// ── Benchmark index ───────────────────────────────────────────────────────────

const benchmarkIndex = new Map<string, BenchmarkEntry>()

for (const b of benchmarks) {
  // Index by metric:niche:region
  for (const region of Object.keys(b.regions)) {
    benchmarkIndex.set(`${b.metric}:${b.niche}:${region}`, b)
    // Wildcard niche fallback (keep first entry per metric:region)
    const wk = `${b.metric}:*:${region}`
    if (!benchmarkIndex.has(wk)) benchmarkIndex.set(wk, b)
  }
}

export function getRegion(country: string): string {
  return valuesets.country_to_region[country.toUpperCase()] ?? 'global'
}

export function buildDistribution(entry: BenchmarkEntry, region: string): Distribution {
  const mean = entry.regions[region] ?? entry.regions.global ?? 0
  const nano = entry.tiers.nano ?? mean * 1.65
  const macro = entry.tiers.macro ?? mean * 0.55

  const p50 = round(mean)
  const p90 = round(nano * 0.95)
  const p10 = round(macro * 0.85)
  const p75 = round(p50 + (p90 - p50) * 0.6)
  const p25 = round(p10 + (p50 - p10) * 0.4)

  return { p10, p25, p50, p75, p90 }
}

function round(n: number) { return Math.round(n * 100) / 100 }

export interface BenchmarkLookupResult {
  entry: BenchmarkEntry
  region: string
  coverage: 'full' | 'partial'
}

export function lookupBenchmark(
  metric: string,
  platform: string,
  country: string,
  niche?: string,
): BenchmarkLookupResult | null {
  // Build full metric key e.g. "ENG.002.IG"
  const parts = metric.split('.')
  const metricKey = parts.length >= 3 ? metric : `${metric}.${platform.toUpperCase()}`
  const region = getRegion(country)

  const tryKey = (k: string, r: string, cov: 'full' | 'partial') => {
    const entry = benchmarkIndex.get(k)
    return entry ? { entry, region: r, coverage: cov } : null
  }

  if (niche) {
    return (
      tryKey(`${metricKey}:${niche}:${region}`, region, 'full') ??
      tryKey(`${metricKey}:${niche}:global`, 'global', 'partial') ??
      tryKey(`${metricKey}:*:${region}`, region, 'partial') ??
      tryKey(`${metricKey}:*:global`, 'global', 'partial') ??
      null
    )
  }

  return (
    tryKey(`${metricKey}:*:${region}`, region, 'partial') ??
    tryKey(`${metricKey}:*:global`, 'global', 'partial') ??
    null
  )
}

// ── ValueSet validation ───────────────────────────────────────────────────────

export const isValidPlatform = (p: string) => valuesets.platforms.includes(p.toUpperCase())
export const isValidFormat   = (f: string) => valuesets.formats.includes(f.toUpperCase())
export const isValidTier     = (t: string) => valuesets.tiers.includes(t.toLowerCase())
export const isValidCountry  = (c: string) => c.toUpperCase() in valuesets.country_to_region
export const getAllMetrics    = () => metrics
```

```typescript
import { Hono } from 'hono'
import { lookupBenchmark, buildDistribution, isValidPlatform, isValidTier, isValidCountry } from '../lib/snomi-index.js'
import { apiError } from '../lib/errors.js'
import type { Verdict } from '../lib/types.js'

const app = new Hono()

function interpolatePercentile(v: number, d: { p10: number; p25: number; p50: number; p75: number; p90: number }): number {
  if (d.p90 <= d.p10) return 50
  if (v >= d.p90) return Math.min(99, 90 + ((v - d.p90) / d.p90) * 9)
  if (v >= d.p75) return 75 + ((v - d.p75) / (d.p90 - d.p75)) * 15
  if (v >= d.p50) return 50 + ((v - d.p50) / (d.p75 - d.p50)) * 25
  if (v >= d.p25) return 25 + ((v - d.p25) / (d.p50 - d.p25)) * 25
  if (v >= d.p10) return 10 + ((v - d.p10) / (d.p25 - d.p10)) * 15
  return Math.max(1, (v / d.p10) * 10)
}

function verdictFromPercentile(p: number): { verdict: Verdict; label: string } {
  const r = Math.round(p)
  if (p >= 90) return { verdict: 'top_performer',  label: `P${r} — top performer` }
  if (p >= 75) return { verdict: 'above_market',   label: `P${r} — au-dessus du marché` }
  if (p >= 25) return { verdict: 'market',          label: `P${r} — dans la norme du marché` }
  if (p >= 10) return { verdict: 'below_market',   label: `P${r} — en dessous du marché` }
  return         { verdict: 'low_performer',        label: `P${r} — faibles performances` }
}

app.get('/', (c) => {
  const metric    = c.req.query('metric')
  const valueStr  = c.req.query('value')
  const platform  = c.req.query('platform')
  const country   = c.req.query('country')
  const tier      = c.req.query('tier')
  const niche     = c.req.query('niche')

  if (!metric || !valueStr || !platform || !country || !tier) {
    return c.json(apiError('MISSING_PARAMETER', "Les paramètres metric, value, platform, country et tier sont requis"), 400)
  }

  const value = parseFloat(valueStr)
  if (isNaN(value)) {
    return c.json(apiError('MISSING_PARAMETER', "Le paramètre 'value' doit être un nombre", 'Exemple: value=5.9'), 400)
  }
  if (!isValidPlatform(platform)) {
    return c.json(apiError('INVALID_DIMENSION', `Plateforme '${platform}' non reconnue`), 400)
  }
  if (!isValidCountry(country)) {
    return c.json(apiError('INVALID_DIMENSION', `Pays '${country}' non reconnu`), 400)
  }
  if (!isValidTier(tier)) {
    return c.json(apiError('INVALID_DIMENSION', `Tier '${tier}' non reconnu`), 400)
  }

  const result = lookupBenchmark(metric, platform, country, niche ?? undefined)
  if (!result) {
    return c.json(apiError('NO_BENCHMARK_DATA', `Aucun benchmark pour ${metric} sur ${platform} en ${country}`), 404)
  }

  const distribution = buildDistribution(result.entry, result.region)
  const percentile   = interpolatePercentile(value, distribution)
  const { verdict, label } = verdictFromPercentile(percentile)
  const parts = metric.split('.')
  const metricKey = parts.length >= 3 ? metric : `${metric}.${platform.toUpperCase()}`

  return c.json({
    metric: `${metricKey}.${country.toUpperCase()}`,
    value,
    percentile: Math.round(percentile),
    verdict,
    label,
    benchmark: { p25: distribution.p25, p50: distribution.p50, p75: distribution.p75, p90: distribution.p90 },
  })
})
```

```typescript
import { Hono } from 'hono'
import { lookupMetric, parseCode } from '../lib/snomi-index.js'
import { apiError } from '../lib/errors.js'
import type { AuditCitation, AuditIssue, RulesLevel, AuditSeverity } from '../lib/types.js'

const app = new Hono()

function auditOne(citation: AuditCitation, rules: RulesLevel): AuditIssue[] {
  const issues: AuditIssue[] = []
  const parsed = parseCode(citation.code)
  const metric = lookupMetric(citation.code)

  if (!metric) {
    issues.push({
      code: citation.code,
      severity: 'error',
      rule: `Le code '${parsed.base}' n'existe pas dans la spec SNOMI v2`,
      fix: 'Vérifier le code sur https://rayanebenabs.github.io/SNOMI',
    })
    return issues
  }

  // Trust ceiling
  if (citation.trust) {
    const declared = parseInt(citation.trust.replace('T', ''))
    const max      = parseInt(metric.trust_level.replace('T', ''))
    if (declared > max) {
      issues.push({
        code: citation.code,
        severity: 'error',
        rule: `Trust déclaré ${citation.trust} dépasse le maximum ${metric.trust_level} pour ${parsed.base}`,
        fix: `Corriger à ${metric.trust_level} ou inférieur`,
      })
    }
  }

  // Undeclared trust
  if (!citation.trust && rules !== 'permissive') {
    const sev: AuditSeverity = rules === 'strict' ? 'error' : 'warning'
    issues.push({
      code: citation.code,
      severity: sev,
      rule: 'Le Trust Score n\'est pas déclaré',
      fix: `Ajouter trust: "${metric.trust_level}" à la citation`,
    })
  }

  // VAL.001 must declare [base: REA.xxx]
  if (parsed.base === 'VAL.001' && rules !== 'permissive') {
    if (!/\[base\s*:/.test(citation.code)) {
      const plat = parsed.platform ?? 'IG'
      const mkt  = parsed.market ?? 'FR'
      issues.push({
        code: citation.code,
        severity: 'error',
        rule: 'VAL.001 doit déclarer la métrique de base: [base: REA.001|REA.002|REA.005|REA.006]',
        fix: `Utiliser: VAL.001[base:REA.001].${plat}.${mkt}`,
      })
    }
  }

  // CON.002 must declare base
  if (parsed.base === 'CON.002' && rules !== 'permissive') {
    if (!/\[base\s*:/.test(citation.code)) {
      const sev: AuditSeverity = rules === 'strict' ? 'error' : 'warning'
      issues.push({
        code: citation.code,
        severity: sev,
        rule: 'CON.002 doit déclarer le dénominateur: [base: REA.001] ou [base: REA.002]',
        fix: `Utiliser: CON.002[base:REA.002]${parsed.platform ? '.' + parsed.platform : ''}`,
      })
    }
  }

  // ENG: market code recommended in strict mode
  if (parsed.domain === 'ENG' && !parsed.market && rules === 'strict') {
    issues.push({
      code: citation.code,
      severity: 'warning',
      rule: 'Le code marché est recommandé en mode strict pour les métriques ENG',
      fix: `Utiliser: ${citation.code}.FR (ou le code pays approprié)`,
    })
  }

  return issues
}

app.post('/', async (c) => {
  let body: { citations?: unknown; rules?: unknown }
  try {
    body = await c.req.json()
  } catch {
    return c.json(apiError('INVALID_BODY', 'Corps de requête JSON invalide'), 400)
  }

  if (!body.citations || !Array.isArray(body.citations)) {
    return c.json(apiError('MISSING_PARAMETER', "Le champ 'citations' (array) est requis"), 400)
  }

  const rules: RulesLevel = (['strict', 'standard', 'permissive'] as const).includes(body.rules as RulesLevel)
    ? (body.rules as RulesLevel)
    : 'standard'

  const citations = body.citations as AuditCitation[]
  const allIssues: AuditIssue[] = []
  const passed: string[] = []

  for (const citation of citations) {
    if (!citation.code) continue
    const issues = auditOne(citation, rules)
    if (!issues.some(i => i.severity === 'error')) passed.push(citation.code)
    allIssues.push(...issues)
  }

  let score = 100
  for (const i of allIssues) {
    if (i.severity === 'error')   score -= 15
    if (i.severity === 'warning') score -= 5
  }
  score = Math.max(0, score)

  return c.json({
    compliant: !allIssues.some(i => i.severity === 'error') && score >= 80,
    score,
    issues: allIssues,
    passed,
  })
})

export default app
```

```typescript
api.route('/compare',   compareRoute)
api.route('/audit',     auditRoute)
app.route('/api/v1', api)

app.notFound((c) => c.json({ error: { code: 'NOT_FOUND', message: 'Endpoint not found' } }, 404))

// Only start server when run directly (not when imported by tests)
if (process.env.NODE_ENV !== 'test') {
  const port = parseInt(process.env.PORT ?? '3000')
  console.log(`SNOMI API running on http://localhost:${port}`)
  serve({ fetch: app.fetch, port })
}
```
